**4. buildPatternContext function (per D-14, D-15):**
```typescript
import type { SupabaseClient } from '@supabase/supabase-js';

export async function buildPatternContext(
  supabase: SupabaseClient,
  userId: string,
): Promise<PatternContext> {
  // Initialize every field to null explicitly — prevents undefined (Pitfall 3)
  const context: PatternContext = {
    attachment_style: null,
    repair_style: null,
    reassurance_need: null,
    space_preference: null,
    stress_communication: null,
    interpretation_bias: null,
    vulnerability_pace: null,
    worth_pattern: null,
  };

  try {
    const { data } = await supabase
      .from('profile_traits')
      .select('trait_key, inferred_value, effective_weight')
      .eq('user_id', userId)
      .in('trait_key', [...PATTERN_KEYS])
      .gte('effective_weight', 0.3);  // D-15: threshold applied at DB layer

    for (const row of data || []) {
      const key = row.trait_key as PatternKey;
      const allowed = VALID_PATTERN_VALUES[key];
      if (!allowed || !allowed.has(row.inferred_value)) continue; // vocab guard per D-09
      context[key] = row.inferred_value;
    }
  } catch {
    // Non-blocking — return all-null context on any error
    // Mirrors existing pattern from peter/morning.ts
  }

  return context;
}
```

Key design points:
- Single Supabase query with `.in('trait_key', [...PATTERN_KEYS])` — not 8 separate queries
- `effective_weight >= 0.3` filter applied at DB layer (per D-15, Pitfall 6)
- Vocabulary guard discards out-of-vocab values (per D-09)
- Never throws — try/catch returns all-null context (per D-13)
- Every field initialized to `null` before query (prevents Pitfall 3)

**5. patternContextToTraits conversion helper (per D-18, solves Research Open Question 1):**

This helper converts `PatternContext` to `ProfileTrait[]` so callers can pass the result to `buildPersonalizedPrompt`. Since `buildPatternContext` already applied the `effective_weight >= 0.3` threshold, the synthesized traits use `confidence: 1.0, effective_weight: 1.0` (these values are not used by `buildPersonalizedPrompt` — it only reads `trait_key` and `inferred_value` to look up `TRAIT_DESCRIPTIONS`).

```typescript
import type { ProfileTrait } from '@/lib/peterService';

export function patternContextToTraits(ctx: PatternContext): ProfileTrait[] {
  const traits: ProfileTrait[] = [];
  for (const key of PATTERN_KEYS) {
    const value = ctx[key];
    if (value !== null) {
      traits.push({
        trait_key: key,
        inferred_value: value,
        confidence: 1.0,
        effective_weight: 1.0,
      });
    }
  }
  return traits;
}
```
